\begin{questions} 
		
	\question Suppose that the number of Krabby Patties sold at the Krusty Krab on a randomly selected day, $X$, follows a normal distribution with a mean of 537 patties and a standard deviation of 38.6 patties.
	
	\vspace{3mm}
	
	\begin{parts}
	
	\part Describe the distribution of $\overline{X}$, the mean number of Krabby Patties sold per day for a random sample of 27 days, by identifying the \textbf{mean} $\mu_{\overline{X}}$ and \textbf{standard error} $\sigma_{\overline{X}}$. Round standard error to four places.
	
	\begin{solution}[\stretch{1}]
	
	\vspace{3mm}

	$\mu_{\overline{X}}=537$ patties \hspace{10mm} $\sigma_{\overline{X}}=\frac{38.6}{\sqrt{27}}=7.4286$ patties
	
	\vspace{3mm}

	\end{solution}	
	
	\part What distribution does $\overline{X}$ follow? How can you tell?
	
	\begin{solution}[\stretch{1}]
	
	\vspace{3mm}

	The population $X$ is normally distributed, so $\overline{X}$ follows an approximately normal distribution.
	
	\vspace{3mm}

	\end{solution}	
	
	\part Suppose that a random sample of 27 days is selected. What is the probability that the sample mean number of Krabby Patties differs from the population mean by \textbf{more than} 15 patties? Include a probability statement, your answer, and an appropriately labeled and shaded sketch.
	
	\begin{solution}[\stretch{1}]
	
	\vspace{3mm}

	Lower value: $537-15=522 \Rightarrow z=\frac{522-537}{7.4286}=-2.02$
	
	\vspace{3mm}
	
	Upper value: $537+15=552 \Rightarrow z=\frac{552-537}{7.4286}=2.02$
	
	\vspace{3mm}
	
	$P\left(\overline{X}<522\right)+P\left(\overline{X}>552\right)=P(Z<-2.02)+P(Z>2.02)=0.0217+0.0217=0.0434$
	
	\vspace{1mm}

	\end{solution}	
	
	\end{parts}
		
	\question A national survey estimated that of all U.S.\ citizens who had visited the hostpital within the last five years, approximately 69\% were satisfied with the treatment received during their visit. In an evaluation of their patient care standards, the staff at Grey-Sloan Memorial Hospital are interested in seeing how their hospital compares to the national result. They plan to interview a sample of 235 randomly selected patients who stayed at Grey-Sloan in the past five years.
	
	\vspace{3mm}
	
	\begin{parts}
	
	\part Describe the distribution of $\hat{p}$, the proportion of satisfied patients in a sample of 235, by describing the \textbf{mean} $\mu_{\hat{p}}$ and \textbf{standard error} $\sigma_{\hat{p}}$. Round standard error to four decimal places.
	
	\begin{solution}[\stretch{1}]
	
	\vspace{3mm}

	$\mu_{\hat{p}}=p=0.69$ \hspace{10mm} $\sigma_{\hat{p}}=\sqrt{\frac{0.69(0.31)}{235}}=0.0302$
	
	\vspace{3mm}

	\end{solution}	
	
	\part What distribution does $\hat{p}$ follow? How can you tell?
	
	\begin{solution}[\stretch{1}]
	
	\vspace{3mm}

	$np=235(0.69)=162.15\geq 5$ and $n(1-p)=235(1-0.69)=72.85\geq 5$
	
	\vspace{3mm}
	
	$\Rightarrow \hat{p}$ follows an approximately normal distribution.
	
	\vspace{3mm}

	\end{solution}	
	
	\part What is the probability that the proportion of satisfied hospital patients at Grey-Sloan Memorial Hospital differs from the national result by \textbf{less than} 5\%? Include a probability statement, your answer, and an appropriately labeled and shaded sketch.
	
	\begin{solution}[\stretch{1}]
	
	\vspace{3mm}

	Lower value: $0.69-0.05=0.64 \Rightarrow z=\frac{0.64-0.69}{0.0302}=-1.66$
	
	\vspace{3mm}
	
	Upper value: $0.69+0.05=0.74 \Rightarrow z=\frac{0.74-0.69}{0.0302}=1.66$
	
	\vspace{3mm}
	
	$P(0.64<\hat{p}<0.74)=P(-1.66<Z<1.66)=0.9031$
	
	\vspace{3mm}

	\end{solution}	
	
	\end{parts}
