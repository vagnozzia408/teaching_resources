	\question Let $X=$ the number of large bags of popcorn sold by a local movie theater in a day. Suppose that $X$ is normally distributed with a mean of 230 bags and a standard deviation of 29 bags. Describe the distribution of $\overline{X}$, the \textbf{average} number of large bags of popcorn sold in a random sample of 7 days, by identifying the following.
	
	\vspace{3mm}
	
	\begin{parts}
	
	\part \textbf{Mean} of the Distribution of $\overline{X}$ (Use the symbol and value on page 117 of the Lecture Guide.)
	
	\begin{solution}[\stretch{1}]
	
	\vspace{3mm}

	$\mu_{\overline{X}}=230$ large popcorn bags
	
	\vspace{3mm}

	\end{solution}	
	
	\part \textbf{Standard Deviation} of the Distribution of $\overline{X}$ (Use the symbol and formula on page 117.)
	
	\begin{solution}[\stretch{1}]
	
	\vspace{3mm}

	$\sigma_{\overline{X}}=\displaystyle \frac{29}{\sqrt{7}}=10.96$ large popcorn bags
	
	\vspace{3mm}

	\end{solution}	
	
	\part \textbf{Shape} of the Distribution of $\overline{X}$ (Use the flowchart on page 118 to determine whether $\overline{X}$ is normally distributed and how you know.)
	
	\begin{solution}[\stretch{1}]
	
	\vspace{3mm}

	$\overline{X}$ is approximately normally distributed because $X$ is normally distributed.
	
	\vspace{3mm}

	\end{solution}	
	
	\end{parts}
		
	\question Suppose that $Y$ describes the number of tickets sold by the movie theater in a day, with a mean of 1040 tickets and a standard deviation of 212 tickets. Describe the distribution of $\overline{Y}$, the \textbf{average} number of tickets sold per day in a random sample of 55 days.
	
	\vspace{3mm}
	
	\begin{parts}
	
	\part \textbf{Mean} of the Distribution of $\overline{Y}$
	
	\begin{solution}[\stretch{1}]
	
	\vspace{3mm}

	$\mu_{\overline{Y}}=1040$ tickets
	
	\vspace{3mm}

	\end{solution}	
	
	\part \textbf{Standard Deviation} of the Distribution of $\overline{Y}$
	
	\begin{solution}[\stretch{1}]
	
	\vspace{3mm}

	$\sigma_{\overline{Y}}=\displaystyle \frac{212}{\sqrt{55}}=28.59$ tickets
	
	\vspace{3mm}

	\end{solution}	
	
	\part \textbf{Shape} of the Distribution of $\overline{Y}$ (i.e.\ whether it's normally distributed and how you know!)
	
	\begin{solution}[\stretch{1}]
	
	\vspace{3mm}

	By the Central Limit Theorem, $\overline{Y}$ is approximately normally distributed because the sample size is large enough ($n=55\geq 30$). 
	
	\vspace{3mm}

	\end{solution}	
	
	\end{parts}

\question Suppose that the number of Krabby Patties sold at the Krusty Krab on a randomly selected day, $X$, follows a normal distribution with a mean of 537 patties and a standard deviation of 38.6 patties.
	
	\vspace{3mm}
	
	\begin{parts}
	
	\part Describe the distribution of $\overline{X}$, the mean number of Krabby Patties sold per day for a random sample of 27 days, by identifying the \textbf{mean} $\mu_{\overline{X}}$ and \textbf{standard error} $\sigma_{\overline{X}}$. Round standard error to four places.
	
	\begin{solution}[\stretch{1}]
	
	\vspace{3mm}

	$\mu_{\overline{X}}=537$ patties \hspace{10mm} $\sigma_{\overline{X}}=\frac{38.6}{\sqrt{27}}=7.4286$ patties
	
	\vspace{3mm}

	\end{solution}	
	
	\part What distribution does $\overline{X}$ follow? How can you tell?
	
	\begin{solution}[\stretch{1}]
	
	\vspace{3mm}

	The population $X$ is normally distributed, so $\overline{X}$ follows an approximately normal distribution.
	
	\vspace{3mm}

	\end{solution}	
	
	\part Suppose that a random sample of 27 days is selected. What is the probability that the sample mean number of Krabby Patties differs from the population mean by \textbf{more than} 15 patties? Include a probability statement, your answer, and an appropriately labeled and shaded sketch.
	
	\begin{solution}[\stretch{1}]
	
	\vspace{3mm}

	Lower value: $537-15=522 \Rightarrow z=\frac{522-537}{7.4286}=-2.02$
	
	\vspace{3mm}
	
	Upper value: $537+15=552 \Rightarrow z=\frac{552-537}{7.4286}=2.02$
	
	\vspace{3mm}
	
	$P\left(\overline{X}<522\right)+P\left(\overline{X}>552\right)=P(Z<-2.02)+P(Z>2.02)=0.0217+0.0217=0.0434$
	
	\vspace{1mm}

	\end{solution}	
	
	\end{parts}
		
	\question A national survey estimated that of all U.S.\ citizens who had visited the hostpital within the last five years, approximately 69\% were satisfied with the treatment received during their visit. In an evaluation of their patient care standards, the staff at Grey-Sloan Memorial Hospital are interested in seeing how their hospital compares to the national result. They plan to interview a sample of 235 randomly selected patients who stayed at Grey-Sloan in the past five years.
	
	\vspace{3mm}
	
	\begin{parts}
	
	\part Describe the distribution of $\hat{p}$, the proportion of satisfied patients in a sample of 235, by describing the \textbf{mean} $\mu_{\hat{p}}$ and \textbf{standard error} $\sigma_{\hat{p}}$. Round standard error to four decimal places.
	
	\begin{solution}[\stretch{1}]
	
	\vspace{3mm}

	$\mu_{\hat{p}}=p=0.69$ \hspace{10mm} $\sigma_{\hat{p}}=\sqrt{\frac{0.69(0.31)}{235}}=0.0302$
	
	\vspace{3mm}

	\end{solution}	
	
	\part What distribution does $\hat{p}$ follow? How can you tell?
	
	\begin{solution}[\stretch{1}]
	
	\vspace{3mm}

	$np=235(0.69)=162.15\geq 5$ and $n(1-p)=235(1-0.69)=72.85\geq 5$
	
	\vspace{3mm}
	
	$\Rightarrow \hat{p}$ follows an approximately normal distribution.
	
	\vspace{3mm}

	\end{solution}	
	
	\part What is the probability that the proportion of satisfied hospital patients at Grey-Sloan Memorial Hospital differs from the national result by \textbf{less than} 5\%? Include a probability statement, your answer, and an appropriately labeled and shaded sketch.
	
	\begin{solution}[\stretch{1}]
	
	\vspace{3mm}

	Lower value: $0.69-0.05=0.64 \Rightarrow z=\frac{0.64-0.69}{0.0302}=-1.66$
	
	\vspace{3mm}
	
	Upper value: $0.69+0.05=0.74 \Rightarrow z=\frac{0.74-0.69}{0.0302}=1.66$
	
	\vspace{3mm}
	
	$P(0.64<\hat{p}<0.74)=P(-1.66<Z<1.66)=0.9031$
	
	\vspace{3mm}

	\end{solution}	
	
	\end{parts}

\question The School of Mathematical and Statistical Sciences has found that for each exam, 3\% of the scantrons are not correctly filled out and the scantron machine will not record a score for those students. Often this is because students forget to bubble in their test version. Suppose that we have a random sample of 300 scantrons from students taking exams in the month of November. Let $p=$ the true proportion of scantrons that are incorrectly completed for exams taken in the month of November, and let $\hat{p}$ be the proportion of incorrectly completed scantrons in our random sample. Use this information to answer the following about the shape of the distribution of $\hat{p}$. 

\vspace{3mm}

\begin{parts}

\part Describe the center or mean of the sampling distribution of $\hat{p}$. (Include the correct symbol in your answer.)

\begin{solution}[\stretch{1}]

\vspace{3mm}

$\mu_{\hat{p}}=p=0.03$

\vspace{3mm}
\end{solution}

\part Describe the spread or standard error of the sampling distribution of $\hat{p}$. Round your answer to four decimal places.

\begin{solution}[\stretch{1}]

\vspace{3mm}

$\sigma_{\hat{p}}=\sqrt{\frac{0.03(0.97)}{300}}=0.0098$

\vspace{3mm}
\end{solution}

\part Describe the shape of the sampling distribution of $\hat{p}$. Justify your reasoning.

\begin{solution}[\stretch{1}]

\vspace{3mm}

The sampling distribution of $\hat{p}$ is approximately normal because $np=9\geq 5$ and $n(1-p)=291\geq 5$. 

\vspace{3mm}
\end{solution}

\end{parts}

\question Steve and Tony are two tenured college professors in Georgia. Rather than spend their Friday evenings complaining about the heat, Steve and Tony sit down to play a game of poker. The winnings for each game have an unknown probability distribution with a mean of \$4 and a standard deviation of \$20. Let $X$ represent the winnings from a  game and let $\overline{X}$ represent the average winnings for a random sample of 35 games.

\vspace{3mm}

\begin{parts}
\part Explain why $\overline{X}$ has a normal distribution.

\begin{solution}[\stretch{1}]

\vspace{3mm}
$\overline{X}$ has a normal distribution because the sample size $n$ is greater than 30.
\vspace{3mm}
\end{solution}

\part Give the mean and standard deviation of $\overline{X}$. Label each value with the appropriate symbol. Round your answer to two decimal places if appropriate.

\begin{solution}[\stretch{1}]
\vspace{3mm}

$\mu_{\overline{X}}=\$4$

\vspace{3mm}

$\sigma_{\overline{X}}=\frac{20}{\sqrt{35}}=\$3.38$
\vspace{3mm}
\end{solution}

\part What is the probability that the average winnings for a random sample of 35 games will be greater than \$0.00? Label your answer with the appropriate probability notation. Round the final answer to four decimal places. To show your work, you may do one of the following: provide a well-labeled sketch with the appropriate area shaded, calculate the $z$-score and use the standard normal table, or write the correct integral for this problem.

\begin{solution}[\stretch{1}]

\vspace{3mm}

$P(\overline{X}>0)=P\left(Z>\frac{0-4}{3.38}\right)=P(Z>-1.18)=0.8817$ if using rounded answer from Part (b) and calculator, $0.8816$ if using calculator and unrounded answer from Part (b), $0.8810$ if using table

\vspace{3mm}

Integral Notation:

\vspace{1mm}

$P(\overline{X}>0)=\int_0^{\infty}\frac{1}{3.38\sqrt{2\pi}}e^{-\frac{(x-4)^2}{2(3.38)^2}}dx$ if using the normal pdf

$P(Z>-1.18)=\int_{-1.18}^{\infty}\frac{1}{\sqrt{2\pi}}e^{-\frac{x^2}{2}}dx$ if using the standard normal pdf

\vspace{3mm}
\end{solution}
\end{parts}

\question It has been determined that 80\% of apartments listed as ``apartment for rent'' on Craigslist have been rented within two weeks of the initial listing. We selected a random sample of 100 ``apartment for rent'' ads on Craigslist and tracked down the rental result for each. We define $\hat{p}$ as the proportion of apartments in our sample of size 100 that rent within two weeks of listing the ad.

\vspace{3mm}

\begin{parts}
\part Is the proportion of apartments that will rent within two weeks in our sample of size 100 normally distributed? Support your response with the appropriate calculation(s).

\begin{solution}[\stretch{1}]

\vspace{3mm}

$np=100(0.80)=80\geq 5$ and $n(1-p)=100(0.20)=20\geq 5 \Rightarrow$ Yes, $\hat{p}$ is normally distributed.

\vspace{3mm}

\end{solution}

\part Find the mean and standard error of the sampling distribution for $\hat{p}$ in this situation.

\begin{solution}[\stretch{1}]

\vspace{3mm}
$\mu_{\hat{p}}=p=0.80$

\vspace{3mm}

$\sigma_{\hat{p}}=\sqrt{\frac{(0.80)(0.20)}{100}}=0.04$
\vspace{3mm}

\end{solution}

\part Find the probability that more than 75 out of 100 ``apartment for rent'' ads will result in renting the unit within two weeks. Give a probability statement, show a z-score calculation, and round your final answer to four places.

\begin{solution}[\stretch{3}]

\vspace{3mm}
$z=\frac{0.75-0.80}{0.04}-1.25$
\vspace{3mm}

$P(\hat{p}>0.75)=P(Z>-1.25)=0.8944$

\end{solution}
\end{parts}
